private static byte[] loadIndexes(Context context) {
        byte[] indexes = null;

        InputStream is = null;
        try {
            is = context.getAssets().open(ASSET);

            indexes = new byte[(END - START + 1) * 2];

            is.read(indexes);
        } catch (Throwable tr) {
            PartnerLogger.d(tr.getMessage());

            indexes = null;
        } finally {
            if (is != null) {
                try {
                    is.close();
                } catch (IOException e) {
                    PartnerLogger.d(e.getMessage());
                }
            }
        }

        return indexes;
    }
